// No MainWindow.xaml.cs — gerar HWID da máquina
private static string GetHWID()
{
  var cpu = new ManagementObjectSearcher("SELECT ProcessorId FROM Win32_Processor");
  var disk = new ManagementObjectSearcher("SELECT SerialNumber FROM Win32_DiskDrive");
  string raw = cpu.Get().Cast<ManagementObject>().First()["ProcessorId"].ToString()
            + disk.Get().Cast<ManagementObject>().First()["SerialNumber"].ToString();
  using var sha = SHA256.Create();
  return Convert.ToHexString(sha.ComputeHash(Encoding.UTF8.GetBytes(raw)))[..16];
}

import discord, requests
from discord.ext import commands

bot = commands.Bot(command_prefix='!')
API = "https://reduxadminkey.netlify.app/api"
ADMIN = "ADMIN_KEY_GERADA_NO_PAINEL"

@bot.command()
async def getkey(ctx, tipo="premium"):
  r = requests.get(f"{API}/generate",
    params={"type": tipo,
      "user": str(ctx.author),
      "token": ADMIN})
  data = r.json()
  if data["success"]:
    await ctx.author.send(
      f"✅ Sua key:\n`{data['key']}`")
    await ctx.send("Key enviada no DM!")
  else:
    await ctx.send("Erro ao gerar key.")

bot.run("SEU_TOKEN_DISCORD")

// redux_bot.js
const { Client, GatewayIntentBits } = require('discord.js')
const axios = require('axios')

const API = "https://reduxadminkey.netlify.app/api"
const ADMIN = "ADMIN_KEY_GERADA_NO_PAINEL"
const TOKEN = "SEU_TOKEN_DISCORD"

const client = new Client({
  intents: [GatewayIntentBits.Guilds,
    GatewayIntentBits.GuildMessages,
    GatewayIntentBits.MessageContent]
})

client.on('messageCreate', async msg => {
  if (!msg.content.startsWith('!')) return
  const [cmd, tipo = 'premium'] = msg.content.slice(1).split(' ')
  if (cmd !== 'getkey') return
  const { data } = await axios.get(`${API}/generate`, {
    params: { type: tipo, user: msg.author.tag, token: ADMIN }
  })
  if (data.success) {
    await msg.author.send(`✅ Sua key:\n\`${data.key}\``)
    await msg.reply('Key enviada no DM!')
  } else {
    await msg.reply('Erro ao gerar key.')
  }
})

client.login(TOKEN)

// Substitui o ValidateKey local por validação via API
private async Task<bool> ValidateKeyOnline(string key)
{
  string hwid = GetHWID(); // função acima
  using var http = new HttpClient();
  var url = $"https://reduxadminkey.netlify.app/api/validate?key={key}&hwid={hwid}";
  var res = await http.GetStringAsync(url);
  var json = System.Text.Json.JsonDocument.Parse(res);
  return json.RootElement.GetProperty("valid").GetBoolean();
}

// Chamar no TryValidateInputKey() antes de HideKeyOverlay
