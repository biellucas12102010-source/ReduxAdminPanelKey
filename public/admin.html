// PYTHON (discord.py)
# pip install aiohttp
import aiohttp
import asyncio

API_URL = "https://seu-site.netlify.app/api"
ADMIN_TOKEN = "redux-admin-secret"

async def generate_key(type: str, user: str) -> str:
  """Gera key automaticamente via API"""
  url = f"{API_URL}/generate?token={ADMIN_TOKEN}&type={type}&user={user}"
  async with aiohttp.ClientSession() as session:
    async with session.get(url) as resp:
      data = await resp.json()
      return data.get("key", "")

# No seu bot:
@client.event
async def on_message(message):
  if "!giveaway" in message.content:
    winner = message.author
    key = await generate_key("premium", str(winner))

    dm = await winner.create_dm()

// JAVASCRIPT (discord.js)
// npm install axios
const axios = require('axios');

const API_URL = 'https://seu-site.netlify.app/api';
const ADMIN_TOKEN = 'redux-admin-secret';

async function generateKey(type, user) {
  const res = await axios.get(`${API_URL}/generate?token=${ADMIN_TOKEN}&type=${type}&user=${encodeURIComponent(user)}`);
  return res.data.success ? res.data.key : null;
}

// Evento de giveaway (use o seu sistema de sorteio)
client.on('messageCreate', async (message) => {
  if (message.content === '!sortear') {
    // Seu código de sorteio aqui...
    const winner = message.mentions.users.first() || message.author;
    const key = await generateKey('premium', winner.username);
    if (key) {
      const dm = await winner.createDM();
      dm.send(`Parabéns! Você ganhou uma key Premium!\\n\\n\\\`\\\`\\\`${key}\\\`\\\`\\\`\\n\\nCopie e use no executor!`);
    }
  }
});

// Função para pagamento confirmado (chame do seu webhook)
async function confirmarPagamento(discordId, tipoKey = 'premium30') {
  const user = client.users.cache.get(discordId);
  if (!user) return;
  const key = await generateKey(tipoKey, user.username);
  if (key) {
    const dm = await user.createDM();
    dm.send(`Pagamento confirmado! Sua key:\\n\\n\\\`\\\`\\\`${key}\\\`\\\`\\\`\\n\\nUse no executor!`);
  }
}

// Exemplo de chamada via webhook de pagamento
app.post('/webhook/pagamento', async (req, res) => {
  const { discord_id, status } = req.body;
  if (status === 'approved') {
    await confirmarPagamento(discord_id, 'premium30');
  }
  res.send('OK');
});
